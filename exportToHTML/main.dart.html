import 'package:flutter/material.dart';
import 'package:local_auth/local_auth.dart';
import 'cadastro_usuario.dart';

void main() {
  runApp(const MyApp());
}

class MyApp extends StatelessWidget {
  const MyApp({super.key});

  @override
  Widget build(BuildContext context) {
    return MaterialApp(
      debugShowCheckedModeBanner: false,
      title: 'Meu App',
      theme: ThemeData(primarySwatch: Colors.blue),
      home: const LoginPage(), // tela inicial
    );
  }
}

class LoginPage extends StatefulWidget {
  const LoginPage({super.key});

  @override
  State<LoginPage> createState() => _LoginPageState();
}

class _LoginPageState extends State<LoginPage> {
  final TextEditingController usuarioController = TextEditingController();
  final TextEditingController senhaController = TextEditingController();
  final LocalAuthentication auth = LocalAuthentication();

  bool hoverEntrar = false;
  bool hoverBio = false;
  bool focusUsuario = false;
  bool focusSenha = false;

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      body: Container(
        // ✅ Fundo em gradiente verde suave
        decoration: const BoxDecoration(
          gradient: LinearGradient(
            colors: [
              Color(0xFFE8F5E9),
              Color(0xFFB2DFDB),
            ],
            begin: Alignment.topCenter,
            end: Alignment.bottomCenter,
          ),
        ),

        child: Column(
          children: [
            // ✅ CONTEÚDO CENTRALIZADO
            Expanded(
              child: Center(
                child: SingleChildScrollView(
                  child: Padding(
                    padding: const EdgeInsets.all(24),
                    child: Column(
                      mainAxisAlignment: MainAxisAlignment.center,
                      crossAxisAlignment: CrossAxisAlignment.center,
                      children: [
                        // ✅ Ícone moderno com fade-in suave
                        AnimatedOpacity(
                          opacity: 1,
                          duration: const Duration(milliseconds: 800),
                          child: const Icon(
                            Icons.account_circle_rounded,
                            size: 90,
                            color: Colors.blue,
                          ),
                        ),

                        const SizedBox(height: 20),

                        // ✅ CAMPO USUÁRIO COM ANIMAÇÃO SUAVE
                        Focus(
                          onFocusChange: (hasFocus) =>
                              setState(() => focusUsuario = hasFocus),
                          child: AnimatedContainer(
                            duration: const Duration(milliseconds: 200),
                            padding: const EdgeInsets.all(2),
                            decoration: BoxDecoration(
                              borderRadius: BorderRadius.circular(22),
                              border: Border.all(
                                color: focusUsuario
                                    ? Colors.green
                                    : Colors.transparent,
                                width: 2,
                              ),
                            ),
                            child: TextField(
                              controller: usuarioController,
                              decoration: InputDecoration(
                                labelText: "Usuário",
                                prefixIcon: const Icon(Icons.person),
                                border: OutlineInputBorder(
                                  borderRadius: BorderRadius.circular(20),
                                ),
                                filled: true,
                                fillColor: Colors.white,
                              ),
                            ),
                          ),
                        ),

                        const SizedBox(height: 20),

                        // ✅ CAMPO SENHA COM ANIMAÇÃO SUAVE
                        Focus(
                          onFocusChange: (hasFocus) =>
                              setState(() => focusSenha = hasFocus),
                          child: AnimatedContainer(
                            duration: const Duration(milliseconds: 200),
                            padding: const EdgeInsets.all(2),
                            decoration: BoxDecoration(
                              borderRadius: BorderRadius.circular(22),
                              border: Border.all(
                                color: focusSenha
                                    ? Colors.green
                                    : Colors.transparent,
                                width: 2,
                              ),
                            ),
                            child: TextField(
                              controller: senhaController,
                              obscureText: true,
                              decoration: InputDecoration(
                                labelText: "Senha",
                                prefixIcon: const Icon(Icons.lock),
                                border: OutlineInputBorder(
                                  borderRadius: BorderRadius.circular(20),
                                ),
                                filled: true,
                                fillColor: Colors.white,
                              ),
                            ),
                          ),
                        ),

                        const SizedBox(height: 30),

                        // ✅ BOTÃO ENTRAR — Outlined + texto verde + expansão no hover
                        MouseRegion(
                          onEnter: (_) => setState(() => hoverEntrar = true),
                          onExit: (_) => setState(() => hoverEntrar = false),
                          child: AnimatedScale(
                            scale: hoverEntrar ? 1.05 : 1.0,
                            duration: const Duration(milliseconds: 200),
                            child: SizedBox(
                              width: double.infinity,
                              child: OutlinedButton(
                                style: OutlinedButton.styleFrom(
                                  foregroundColor: Colors.green,
                                  padding: const EdgeInsets.symmetric(vertical: 14),
                                  side: const BorderSide(
                                      color: Colors.green, width: 2),
                                  shape: RoundedRectangleBorder(
                                    borderRadius: BorderRadius.circular(25),
                                  ),
                                ),
                                onPressed: () {},
                                child: const Text(
                                  "Entrar",
                                  style: TextStyle(
                                    fontSize: 16,
                                    fontWeight: FontWeight.bold,
                                    color: Colors.green,
                                  ),
                                ),
                              ),
                            ),
                          ),
                        ),

                        const SizedBox(height: 20),

                        // ✅ BOTÃO BIOMETRIA — Outlined + texto verde + expansão no hover
                        MouseRegion(
                          onEnter: (_) => setState(() => hoverBio = true),
                          onExit: (_) => setState(() => hoverBio = false),
                          child: AnimatedScale(
                            scale: hoverBio ? 1.05 : 1.0,
                            duration: const Duration(milliseconds: 200),
                            child: SizedBox(
                              width: double.infinity,
                              child: OutlinedButton.icon(
                                style: OutlinedButton.styleFrom(
                                  foregroundColor: Colors.green,
                                  padding: const EdgeInsets.symmetric(vertical: 14),
                                  side: const BorderSide(
                                      color: Colors.green, width: 2),
                                  shape: RoundedRectangleBorder(
                                    borderRadius: BorderRadius.circular(25),
                                  ),
                                ),
                                onPressed: () {},
                                icon: const Icon(Icons.fingerprint,
                                    size: 28, color: Colors.green),
                                label: const Text(
                                  "Acessar com biometria",
                                  style: TextStyle(
                                    fontSize: 16,
                                    fontWeight: FontWeight.bold,
                                    color: Colors.green,
                                  ),
                                ),
                              ),
                            ),
                          ),
                        ),

                        const SizedBox(height: 20),

                        // ✅ Botão Criar novo usuário (permanece azul)
                        SizedBox(
                          width: double.infinity,
                          child: OutlinedButton(
                            style: OutlinedButton.styleFrom(
                              foregroundColor: Colors.blue.shade700,
                              padding: const EdgeInsets.symmetric(vertical: 14),
                              side: BorderSide(
                                  color: Colors.blue.shade700, width: 2),
                              shape: RoundedRectangleBorder(
                                borderRadius: BorderRadius.circular(25),
                              ),
                            ),
                            onPressed: () {
                              Navigator.push(
                                context,
                                MaterialPageRoute(builder: (_) => const CadastroUsuarioPage()),
                              );
                            },
                            child: const Text(
                              "Criar usuário",
                              style: TextStyle(
                                  fontSize: 16, fontWeight: FontWeight.bold),
                            ),
                          ),
                        ),
                      ],
                    ),
                  ),
                ),
              ),
            ),

            // ✅ RODAPÉ FIXO
            const Padding(
              padding: EdgeInsets.only(bottom: 20),
              child: Text(
                "O seu app de controle de medicamentos",
                style: TextStyle(
                  fontSize: 16,
                  fontWeight: FontWeight.w600,
                  color: Colors.black54,
                ),
                textAlign: TextAlign.center,
              ),
            ),
          ],
        ),
      ),
    );
  }
}
