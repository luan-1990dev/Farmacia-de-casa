import 'package:flutter/material.dart';
import 'login_page.dart';

class CadastroUsuarioPage extends StatefulWidget {
  const CadastroUsuarioPage({super.key});

  @override
  State<CadastroUsuarioPage> createState() => _CadastroUsuarioPageState();
}

class _CadastroUsuarioPageState extends State<CadastroUsuarioPage> {
  final TextEditingController nomeController = TextEditingController();
  final TextEditingController emailController = TextEditingController();
  final TextEditingController senhaController = TextEditingController();

  bool focusNome = false;
  bool focusEmail = false;
  bool focusSenha = false;
  bool hoverCriar = false;

  String? erroNome;
  String? erroEmail;
  String? erroSenha;

  Future<void> validarCampos() async {
    setState(() {
      erroNome = nomeController.text.trim().isEmpty
          ? "Digite seu nome completo"
          : null;

      erroEmail = emailController.text.trim().isEmpty
          ? "Digite seu e-mail"
          : (!emailController.text.contains("@") ||
          !emailController.text.contains("."))
          ? "E-mail inválido"
          : null;

      erroSenha = senhaController.text.trim().isEmpty
          ? "Digite uma senha"
          : senhaController.text.length < 6
          ? "A senha deve ter pelo menos 6 caracteres"
          : null;
    });

    // Se houver erros, não continua
    if (erroNome != null || erroEmail != null || erroSenha != null) {
      return;
    }

    // ✅ SnackBar com animação + fundo verde vibrante
    ScaffoldMessenger.of(context).showSnackBar(
      SnackBar(
        duration: const Duration(seconds: 05),
        backgroundColor: Colors.transparent,
        elevation: 0,
        behavior: SnackBarBehavior.floating,
        content: TweenAnimationBuilder<double>(
          tween: Tween(begin: 0, end: 1),
          duration: const Duration(milliseconds: 500),
          builder: (context, value, child) {
            return Opacity(
              opacity: value,
              child: Transform.translate(
                offset: Offset(0, 20 * (1 - value)),
                child: Container(
                  padding: const EdgeInsets.all(16),
                  decoration: BoxDecoration(
                    color: Colors.green, // ✅ verde vibrante
                    borderRadius: BorderRadius.circular(16),
                    boxShadow: const [
                      BoxShadow(
                        color: Colors.black26,
                        blurRadius: 8,
                        offset: Offset(0, 4),
                      ),
                    ],
                  ),
                  child: Row(
                    children: const [
                      Icon(Icons.check_circle, color: Colors.white, size: 28),
                      SizedBox(width: 12),
                      Expanded(
                        child: Text(
                          "Usuário cadastrado com sucesso!",
                          style: TextStyle(
                            color: Colors.white,
                            fontSize: 16,
                            fontWeight: FontWeight.bold,
                          ),
                        ),
                      ),
                    ],
                  ),
                ),
              ),
            );
          },
        ),
      ),
    );

    // ✅ Aguarda os 10 segundos
    await Future.delayed(const Duration(seconds: 05));

    // ✅ Retorna para a tela inicial de login
    Navigator.pushReplacement(
      context,
      MaterialPageRoute(builder: (_) => const LoginPage()),
    );
  }

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(
        title: const Text("Cadastro de Usuário"),
        centerTitle: true,
        backgroundColor: Colors.blue.shade600,
      ),

      body: Container(
        decoration: const BoxDecoration(
          gradient: LinearGradient(
            colors: [
              Color(0xFFE3F2FD),
              Color(0xFFBBDEFB),
            ],
            begin: Alignment.topCenter,
            end: Alignment.bottomCenter,
          ),
        ),

        child: Center(
          child: SingleChildScrollView(
            child: Padding(
              padding: const EdgeInsets.all(24),
              child: Column(
                crossAxisAlignment: CrossAxisAlignment.center,
                children: [
                  const Icon(
                    Icons.person_add_rounded,
                    size: 90,
                    color: Colors.blue,
                  ),

                  const SizedBox(height: 30),

                  // ✅ Campo Nome
                  Focus(
                    onFocusChange: (hasFocus) =>
                        setState(() => focusNome = hasFocus),
                    child: AnimatedContainer(
                      duration: const Duration(milliseconds: 200),
                      padding: const EdgeInsets.all(2),
                      decoration: BoxDecoration(
                        borderRadius: BorderRadius.circular(22),
                        border: Border.all(
                          color: erroNome != null
                              ? Colors.red
                              : focusNome
                              ? Colors.blue
                              : Colors.transparent,
                          width: 2,
                        ),
                      ),
                      child: TextField(
                        controller: nomeController,
                        decoration: InputDecoration(
                          labelText: "Nome completo",
                          errorText: erroNome,
                          prefixIcon: const Icon(Icons.person),
                          border: OutlineInputBorder(
                            borderRadius: BorderRadius.circular(20),
                          ),
                          filled: true,
                          fillColor: Colors.white,
                        ),
                      ),
                    ),
                  ),

                  const SizedBox(height: 20),

                  // ✅ Campo Email
                  Focus(
                    onFocusChange: (hasFocus) =>
                        setState(() => focusEmail = hasFocus),
                    child: AnimatedContainer(
                      duration: const Duration(milliseconds: 200),
                      padding: const EdgeInsets.all(2),
                      decoration: BoxDecoration(
                        borderRadius: BorderRadius.circular(22),
                        border: Border.all(
                          color: erroEmail != null
                              ? Colors.red
                              : focusEmail
                              ? Colors.blue
                              : Colors.transparent,
                          width: 2,
                        ),
                      ),
                      child: TextField(
                        controller: emailController,
                        decoration: InputDecoration(
                          labelText: "E-mail",
                          errorText: erroEmail,
                          prefixIcon: const Icon(Icons.email),
                          border: OutlineInputBorder(
                            borderRadius: BorderRadius.circular(20),
                          ),
                          filled: true,
                          fillColor: Colors.white,
                        ),
                      ),
                    ),
                  ),

                  const SizedBox(height: 20),

                  // ✅ Campo Senha
                  Focus(
                    onFocusChange: (hasFocus) =>
                        setState(() => focusSenha = hasFocus),
                    child: AnimatedContainer(
                      duration: const Duration(milliseconds: 200),
                      padding: const EdgeInsets.all(2),
                      decoration: BoxDecoration(
                        borderRadius: BorderRadius.circular(22),
                        border: Border.all(
                          color: erroSenha != null
                              ? Colors.red
                              : focusSenha
                              ? Colors.blue
                              : Colors.transparent,
                          width: 2,
                        ),
                      ),
                      child: TextField(
                        controller: senhaController,
                        obscureText: true,
                        decoration: InputDecoration(
                          labelText: "Senha",
                          errorText: erroSenha,
                          prefixIcon: const Icon(Icons.lock),
                          border: OutlineInputBorder(
                            borderRadius: BorderRadius.circular(20),
                          ),
                          filled: true,
                          fillColor: Colors.white,
                        ),
                      ),
                    ),
                  ),

                  const SizedBox(height: 30),

                  // ✅ Botão Criar Conta
                  MouseRegion(
                    onEnter: (_) => setState(() => hoverCriar = true),
                    onExit: (_) => setState(() => hoverCriar = false),
                    child: AnimatedScale(
                      scale: hoverCriar ? 1.05 : 1.0,
                      duration: const Duration(milliseconds: 200),
                      child: SizedBox(
                        width: double.infinity,
                        child: OutlinedButton(
                          style: OutlinedButton.styleFrom(
                            foregroundColor: Colors.blue.shade700,
                            padding: const EdgeInsets.symmetric(vertical: 14),
                            side: BorderSide(
                              color: Colors.blue.shade700,
                              width: 2,
                            ),
                            shape: RoundedRectangleBorder(
                              borderRadius: BorderRadius.circular(25),
                            ),
                          ),
                          onPressed: validarCampos,
                          child: const Text(
                            "Criar conta",
                            style: TextStyle(
                              fontSize: 16,
                              fontWeight: FontWeight.bold,
                            ),
                          ),
                        ),
                      ),
                    ),
                  ),
                ],
              ),
            ),
          ),
        ),
      ),
    );
  }
}
