import 'package:crtl_02/home_page.dart';
import 'package:flutter/material.dart';
import 'adicionar_medicamentos_page.dart';
import 'dower_page.dart';

void main() {
  runApp(const MyApp());
}

class MyApp extends StatelessWidget {
  const MyApp({super.key});

  @override
  Widget build(BuildContext context) {
    return MaterialApp(
      debugShowCheckedModeBanner: false,
      title: "Farmácia de Casa",
      theme: ThemeData(primarySwatch: Colors.blue),
      home: const HomePage(),
    // rota inicial
    initialRoute: '/',
    routes: {
    '/': (context) => const AdicionarMedicamentosPage(),
    '/dower': (context) => const DowerPage(),
    },
   );
  }
}

// Página principal com Drawer
class NavigationDrawerPage extends StatelessWidget {
  const NavigationDrawerPage({super.key});

  @override
  Widget build(BuildContext context) {
    final screenWidth = MediaQuery.of(context).size.width;
    final screenHeight = MediaQuery.of(context).size.height;

    return Scaffold(
      appBar: AppBar(
        title: const Text("Menu de Navegação"),
        centerTitle: true,
        backgroundColor: Colors.blue.shade600,
      ),
      drawer: Drawer(
        child: ListView(
          padding: EdgeInsets.zero,
          children: [
            const DrawerHeader(
              decoration: BoxDecoration(color: Colors.blue),
              child: Center(
                child: Text(
                  "Farmácia de Casa",
                  style: TextStyle(
                    color: Colors.white,
                    fontSize: 22,
                    fontWeight: FontWeight.bold,
                  ),
                ),
              ),
            ),

            // Continuação da sequência
            ListTile(
              leading: const Icon(Icons.medical_services, color: Colors.blue),
              title: const Text("Adicionar Medicamentos"),
              onTap: () {
                Navigator.push(
                  context,
                  MaterialPageRoute(builder: (_) => const AdicionarMedicamentosPage()),
                );
              },
            ),
            ListTile(
              leading: const Icon(Icons.medical_services, color: Colors.blue),
              title: const Text("Vias de Aplicação"),
              onTap: () {
                Navigator.push(
                  context,
                  MaterialPageRoute(builder: (_) => const ViasAplicacaoPage()),
                );
              },
            ),
            ListTile(
              leading: const Icon(Icons.assignment, color: Colors.blue),
              title: const Text("Exames"),
              onTap: () {
                Navigator.push(
                  context,
                  MaterialPageRoute(builder: (_) => const ExamesPage()),
                );
              },
            ),
            ListTile(
              leading: const Icon(Icons.people, color: Colors.blue),
              title: const Text("Residentes"),
              onTap: () {
                Navigator.push(
                  context,
                  MaterialPageRoute(builder: (_) => const ResidentesPage()),
                );
              },
            ),
            ListTile(
              leading: const Icon(Icons.favorite, color: Colors.blue),
              title: const Text("Sua Saúde"),
              onTap: () {
                Navigator.push(
                  context,
                  MaterialPageRoute(builder: (_) => const SuaSaudePage()),
                );
              },
            ),
            ListTile(
              leading: const Icon(Icons.contact_phone, color: Colors.blue),
              title: const Text("Contatos"),
              onTap: () {
                Navigator.push(
                  context,
                  MaterialPageRoute(builder: (_) => const ContatosPage()),
                );
              },
            ),
            const Divider(),
            ListTile(
              leading: const Icon(Icons.logout, color: Colors.red),
              title: const Text("Sair"),
              onTap: () {
                Navigator.pushReplacement(
                  context,
                  MaterialPageRoute(builder: (_) => const LoginPage()),
                );
              },
            ),
          ],
        ),
      ),
      body: Container(
        color: const Color(0xFFBBDEFB),
        child: Center(
          child: Padding(
            padding: const EdgeInsets.all(16.0),
            child: SizedBox(
              width: screenWidth * 0.9,
              height: screenHeight * 0.75,
              child: Image.asset(
                //'assets/farmacia_de_casa.jpeg',//
                fit: BoxFit.contain,
              ),
            ),
          ),
        ),
      ),
    );
  }
}
